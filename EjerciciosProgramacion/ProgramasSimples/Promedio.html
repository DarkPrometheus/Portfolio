        public static void Promedio()
        {
            string[] stringValues = new string[4] {"primera", "segunda", "tercera", "cuarta"};
            int[] calificaciones = new int[4];

            for (int i = 0; i < 4; i++)
            {
                Console.Write(String.Format($"Ingresa la {stringValues[i]} calificacion: "));
                string input = Console.ReadLine();

                bool done = false;
                while (!done)
                {
                    if (IsInt(input))
                    {
                        done = true;
                        calificaciones[i] = int.Parse(input);
                    }
                    else
                    {
                        Console.WriteLine("Debes ingresar un valor numerico: ");
                        input = Console.ReadLine();
                        Console.Clear();
                    }
                }
            }
            Console.Clear();

            int suma = 0;
            for (int i = 0; i < 4; i++)
                suma += calificaciones[i];

            for (int i = 0; i < 4; i++)
                Console.WriteLine(String.Format($"{stringValues[i]} calificacion: ") + calificaciones[i]);
            Console.WriteLine("Promedio: " + ((double)suma / 4));
        }

        public static bool IsInt(string input)
        {
            int value;
            if (int.TryParse(input, out value))
                return true;
            else
                return false;
        }
